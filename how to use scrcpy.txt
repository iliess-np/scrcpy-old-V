	.\scrcpy.exe --no-control
	.\scrcpy.exe -n

	.\scrcpy.exe --stay-awake
	.\scrcpy.exe -w
	.\scrcpy.exe --turn-screen-off --stay-awake
	.\scrcpy.exe -Sw

//Show touches
	.\scrcpy.exe --show-touches
	.\scrcpy.exe -t

	.\scrcpy.exe --turn-screen-off
	.\scrcpy.exe -S
		**//To turn it back on, press 
			Ctrl+Shift+o 
			(or POWER, Ctrl+p).

//Lock video orientation To lock the orientation of the mirroring  => Ctrl+← (left) and Ctrl+→ (right). => Ctrl+r switch between portrait and landscape 
	.\scrcpy.exe --lock-video-orientation 0   # natural orientation
	.\scrcpy.exe --lock-video-orientation 1   # 90° counterclockwise
	.\scrcpy.exe --lock-video-orientation 2   # 180°
	.\scrcpy.exe --lock-video-orientation 3   # 90° clockwise

//Crop to mirror only part of the screen. (If --max-size is also specified, resizing is applied after cropping.)
	.\scrcpy.exe --crop 1224:1440:0:0   # 1224x1440 at offset (0,0)

//full screen
	.\scrcpy.exe -f

//limit both the width and height to some value 
	.\scrcpy.exe --max-size 1024
	.\scrcpy.exe -m 1024  # short version

//Change bit-rate
	.\scrcpy.exe --bit-rate 2M
	.\scrcpy.exe -b 2M  # short version
	
//Limit frame rate  (supported since Android 10)
	.\scrcpy.exe --max-fps 15

//for wireless
	.\adb.exe tcpip 5555
	.\adb.exe connect 192.168.1.5:5555
	.\scrcpy.exe
	
	**//Multi-devices
		**If several devices are listed in adb devices, you must specify the serial:

		**If the device is connected over TCP/IP:
			.\scrcpy.exe --serial 192.168.0.1:5555
			.\scrcpy.exe -s 192.168.1.5:5555  # short versio
